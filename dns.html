DNS Server Installation • BIND9 on Ubuntu
STEP 1

Static IP Configuration (Netplan)

sudo nano /etc/netplan/00-installer-config.yaml
network:
  version: 2
  ethernets:
    enp0s3:
      dhcp4: false
    enp0s8:
      addresses: [192.168.129.172/24]
      nameservers:
        addresses: [1.1.1.1, 8.8.8.8]

sudo netplan apply
STEP 2

Install BIND9

sudo apt update
sudo apt install bind9 bind9-utils -y
STEP 3

Enable Firewall

sudo ufw enable
sudo ufw allow bind9
sudo ufw status
STEP 4

Start BIND9 Service

sudo systemctl start bind9
sudo systemctl status bind9
STEP 5

Main DNS Configuration

sudo nano /etc/bind/named.conf.options

options {
directory "/var/cache/bind";
listen-on { any; };
allow-query { localhost; 192.168.129.0/24; };

forwarders {
8.8.8.8;
};

dnssec-validation no;
};
STEP 6

Force IPv4

sudo nano /etc/default/named

OPTIONS="-u bind -4"
sudo named-checkconf
sudo systemctl restart bind9
STEP 7

Declare DNS Zones

sudo nano /etc/bind/named.conf.local

zone "diarabaka.local" IN {
type master;
file "/etc/bind/zones/db.diarabaka.local";
};

zone "129.168.192.in-addr.arpa" {
type master;
file "/etc/bind/zones/db.129.168.192";
};
STEP 8

Create DNS Zone Files

sudo mkdir -p /etc/bind/zones
sudo cp /etc/bind/db.local /etc/bind/zones/db.diarabaka.local
STEP 9

DNS Testing

host dns01.diarabaka.local
ping dns01.diarabaka.local
host dns01
host 192.168.129.172
STEP 10

Security

sudo chown -R bind:bind /etc/bind/zones
